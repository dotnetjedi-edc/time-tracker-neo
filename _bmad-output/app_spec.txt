<project_specification>
  <project_name>Time Tracker</project_name>

  <overview>
    Time Tracker est une application web MVP minimaliste de suivi de temps personnel. Elle permet de démarrer/arrêter des chronomètres via de gros boutons cliquables organisés en grille, d'organiser les tâches avec des tags, et de visualiser un rapport hebdomadaire du temps passé sur chaque tâche. L'application cible principalement l'utilisation sur tablette, mais reste responsive pour mobile et desktop.
  </overview>

  <technology_stack>
    <frontend>
      <framework>React</framework>
      <build_tool>Vite</build_tool>
      <styling>Tailwind CSS</styling>
      <state_management>React Context API ou Zustand (pour l'état global des chronomètres)</state_management>
    </frontend>
    <backend>
      <runtime>Node.js</runtime>
      <deployment>Vercel Serverless Functions</deployment>
      <database>Turso (libSQL/SQLite distribuée)</database>
      <orm>Drizzle ORM (recommandé pour Turso)</orm>
    </backend>
    <communication>
      <api>REST API</api>
      <data_format>JSON</data_format>
    </communication>
  </technology_stack>

  <core_features>
    <task_management>
      - Créer une nouvelle tâche avec nom
      - Renommer une tâche existante
      - Supprimer une tâche
      - Ajouter un commentaire optionnel à une tâche (via bouton options)
      - Modifier le commentaire d'une tâche
      - Supprimer le commentaire d'une tâche
      - Assigner un ou plusieurs tags à une tâche
      - Retirer des tags d'une tâche
      - Afficher la liste des tâches sous forme de grille de gros boutons
      - Réorganiser les tâches par drag & drop dans la grille
      - Sauvegarder l'ordre des tâches
      - État vide au premier démarrage : affichage du bouton "Nouvelle tâche" uniquement
    </task_management>

    <timer_functionality>
      - Démarrer un chronomètre en cliquant sur le bouton d'une tâche
      - Arrêter le chronomètre actif en re-cliquant sur le même bouton
      - Arrêter automatiquement tout autre chronomètre quand un nouveau démarre (un seul actif à la fois)
      - Afficher le temps accumulé sur chaque bouton de tâche
      - Mettre à jour l'affichage du temps en temps réel (seconde par seconde) pour le chronomètre actif
      - Animation visuelle du bouton actif (pulsation ou effet visuel)
      - Indication visuelle claire du chronomètre actif vs inactifs
      - Arrêter automatiquement le chronomètre si l'application est fermée/rechargée
      - Enregistrer le temps écoulé dans la base de données à chaque arrêt
      - Accumuler le temps pour chaque session de travail sur une tâche
      - Afficher le temps total accumulé pour chaque tâche (toutes sessions confondues)
      - Format d'affichage du temps : HH:MM:SS
    </timer_functionality>

    <tag_system>
      - Créer un nouveau tag avec nom et couleur
      - Supprimer un tag existant
      - Renommer un tag
      - Changer la couleur d'un tag
      - Afficher tous les tags disponibles dans un menu dédié
      - Assigner un tag à une tâche en un clic
      - Permettre plusieurs tags par tâche
      - Afficher les tags associés à chaque tâche
      - Gérer les tâches orphelines quand un tag est supprimé (retirer l'association)
      - Filtrer l'affichage par tag(s) sélectionné(s)
    </tag_system>

    <calendar_view>
      - Afficher une vue calendrier hebdomadaire
      - Navigation entre les semaines (semaine précédente/suivante)
      - Afficher le temps passé sur chaque tâche pour chaque jour de la semaine
      - Calculer et afficher le total d'heures par jour
      - Calculer et afficher le total d'heures pour la semaine complète
      - Filtrer la vue calendrier par tag(s)
      - Afficher les tags associés aux tâches dans le calendrier
      - Afficher les jours de la semaine (Lun-Dim)
      - Mettre en évidence le jour actuel
      - Afficher les données en format heures:minutes (ex: 2h 30m)
      - Gérer l'état vide (semaine sans données enregistrées)
      - Basculer entre vue principale (grille de boutons) et vue calendrier
    </calendar_view>

    <responsive_design>
      - Layout optimisé pour tablette (cas d'usage principal)
      - Adaptation responsive pour mobile (smartphone)
      - Adaptation responsive pour desktop (ordinateur)
      - Taille des boutons adaptée à l'écran (grands et facilement cliquables)
      - Grille responsive (nombre de colonnes adapté à la largeur d'écran)
      - Touch-friendly pour drag & drop sur tablette/mobile
      - Navigation adaptée à chaque taille d'écran
      - Texte lisible sur toutes les tailles d'écran
    </responsive_design>
  </core_features>

  <database_schema>
    <tables>
      <tasks>
        - id (INTEGER PRIMARY KEY AUTOINCREMENT)
        - name (TEXT NOT NULL)
        - comment (TEXT NULL)
        - total_time_seconds (INTEGER DEFAULT 0) - temps total accumulé
        - position (INTEGER DEFAULT 0) - ordre dans la grille
        - created_at (DATETIME DEFAULT CURRENT_TIMESTAMP)
        - updated_at (DATETIME DEFAULT CURRENT_TIMESTAMP)
      </tasks>

      <tags>
        - id (INTEGER PRIMARY KEY AUTOINCREMENT)
        - name (TEXT NOT NULL UNIQUE)
        - color (TEXT NOT NULL) - code couleur hex
        - created_at (DATETIME DEFAULT CURRENT_TIMESTAMP)
      </tags>

      <task_tags>
        - task_id (INTEGER REFERENCES tasks(id) ON DELETE CASCADE)
        - tag_id (INTEGER REFERENCES tags(id) ON DELETE CASCADE)
        - PRIMARY KEY (task_id, tag_id)
      </task_tags>

      <time_entries>
        - id (INTEGER PRIMARY KEY AUTOINCREMENT)
        - task_id (INTEGER REFERENCES tasks(id) ON DELETE CASCADE)
        - start_time (DATETIME NOT NULL)
        - end_time (DATETIME NULL)
        - duration_seconds (INTEGER DEFAULT 0)
        - date (DATE NOT NULL) - date de l'entrée pour les rapports
        - created_at (DATETIME DEFAULT CURRENT_TIMESTAMP)
      </time_entries>

      <active_timer>
        - id (INTEGER PRIMARY KEY CHECK (id = 1)) - une seule ligne
        - task_id (INTEGER REFERENCES tasks(id) ON DELETE SET NULL)
        - start_time (DATETIME)
        - updated_at (DATETIME DEFAULT CURRENT_TIMESTAMP)
      </active_timer>
    </tables>
  </database_schema>

  <api_endpoints_summary>
    <tasks>
      - GET /api/tasks - Récupérer toutes les tâches avec leurs tags
      - POST /api/tasks - Créer une nouvelle tâche
      - PATCH /api/tasks/:id - Modifier une tâche (nom, commentaire)
      - DELETE /api/tasks/:id - Supprimer une tâche
      - PATCH /api/tasks/reorder - Réorganiser les tâches (drag & drop)
    </tasks>

    <tags>
      - GET /api/tags - Récupérer tous les tags
      - POST /api/tags - Créer un nouveau tag
      - PATCH /api/tags/:id - Modifier un tag (nom, couleur)
      - DELETE /api/tags/:id - Supprimer un tag
      - POST /api/tasks/:id/tags - Assigner un tag à une tâche
      - DELETE /api/tasks/:id/tags/:tagId - Retirer un tag d'une tâche
    </tags>

    <timers>
      - GET /api/timer/active - Récupérer le chronomètre actif
      - POST /api/timer/start - Démarrer un chronomètre (arrête les autres)
      - POST /api/timer/stop - Arrêter le chronomètre actif
      - GET /api/tasks/:id/time - Obtenir le temps total d'une tâche
    </timers>

    <reports>
      - GET /api/reports/weekly?date=YYYY-MM-DD - Rapport hebdomadaire
      - GET /api/reports/weekly?date=YYYY-MM-DD&tags=id1,id2 - Rapport filtré par tags
      - GET /api/reports/daily?date=YYYY-MM-DD - Rapport journalier
    </reports>
  </api_endpoints_summary>

  <ui_layout>
    <main_structure>
      Layout principal avec deux vues :

      1. **Vue Principale (Grille de tâches)** :
         - Header : Titre "Time Tracker" + bouton bascule vers Calendrier + bouton Gérer Tags
         - Grille responsive de gros boutons de tâches (2-4 colonnes selon écran)
         - Chaque bouton affiche : nom de la tâche, temps total, tags associés, icône options
         - Bouton "+" (Nouvelle tâche) en bas ou en dernière position de la grille
         - Animation/highlight visuel pour le bouton avec chronomètre actif

      2. **Vue Calendrier Hebdomadaire** :
         - Header : Titre + Navigation semaine (← →) + Filtres tags + bouton retour à Grille
         - Calendrier : 7 colonnes (Lun-Dim) avec liste des tâches et temps par jour
         - Footer : Total par jour sous chaque colonne + Total semaine global
         - Mise en évidence du jour actuel

      3. **Modales/Overlays** :
         - Formulaire "Nouvelle tâche" (nom, tags optionnels)
         - Menu options tâche (renommer, commentaire, supprimer)
         - Menu gestion des tags (CRUD tags)
         - Filtres par tags (multi-sélection)
    </main_structure>

    <responsive_behavior>

    </responsive_behavior>
  </ui_layout>
